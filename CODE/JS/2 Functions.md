### Named Function
```js
// Named Function
function rocketToMars() {
  return 'BOOM!';
}
```

### Anonymous Function
```js
// Anonymous Function
const rocketToMars = function() {
  return 'BOOM!';
}
```

### Function Expressions
```js
// Function Expressions
const dog = function() {
  return 'Woof!';
}
```
